#include <stdio.h>                                                                                                                                                                                                       
#include <stdlib.h>                                                                                                                                                                                                      
                                                                                                                                                                                                                         
int main(int argc, char* argv[], char* envp[]) {                                                                                                                                                                         
    FILE *fp;                                                                                                                                                                                                            
    int i=0;                                                                                                                                                                                                             
    fp=fopen("lab5.txt","a");                                                                                                                                                                                            
    printf("logging %d argument(s)",argc);                                                                                                                                                                               
    for(i=0;i<argc;i++) {                                                                                                                                                                                                
            fprintf(fp,"Argument %d : %s\n",i,argv[i]);                                                                                                                                                                  
    }                                                                                                                                                                                                                    
    for(i=0;envp[i]!=NULL;i++) {                                                                                                                                                                                         
            fprintf(fp,"Environment:%s\n",envp[i]);                                                                                                                                                                      
    }                                                                                                                                                                                                                    
    printf("and %d environment variables\n",i);                                                                                                                                                                          
    fclose(fp);                                                                                                                                                                                                          
}
